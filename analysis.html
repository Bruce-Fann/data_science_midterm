Numerically coded categorical variables are converted to factors. The patient ID column is excluded from modeling.

factor_vars <- c("menopause", "race", "tumor_grade",
                 "ER", "PR", "HER2", "stage",
                 "surgery", "chemo", "hormonal_therapy")

recode_factors <- function(df) {
  df |> mutate(across(all_of(factor_vars), as.factor))
}

dat_a <- recode_factors(dat_a)
dat_b <- recode_factors(dat_b)

# Predictor matrices and outcome vectors
x_a <- dat_a |> select(-id, -log_PFS)
y_a <- dat_a$log_PFS
x_b <- dat_b |> select(-id, -log_PFS)
y_b <- dat_b$log_PFS

missing_tbl <- dat_a |>

Continuous Predictors

cont_vars <- c("age", "bmi", "tumor_size", "lymph_nodes", "Ki67")

dat_a |>
  select(all_of(cont_vars), log_PFS) |>
  pivot_longer(-log_PFS, names_to = "variable", values_to = "value") |>
  ggplot(aes(x = value, y = log_PFS)) +
  geom_point(alpha = 0.25, size = 0.7, color = "steelblue") +
  geom_smooth(method = "loess", se = TRUE,
              color = "firebrick", linewidth = 0.8) +
  facet_wrap(~ variable, scales = "free_x", ncol = 3) +
  labs(title = "Continuous Predictors vs. log(PFS)",
       x = "Predictor value", y = "log(PFS)") +
  theme_bw()
